    The test uses a synthetic VariableStore (no-op) and a TrackingVariableStore
    wrapping it, measuring raw get() call overhead across 10_000 iterations.

    The overhead is measured at the TrackingVariableStore layer directly (not via
    a full tree build) to isolate the tracking cost from tree construction noise:

    ```rust
    #[test]
    fn service_field_tracking_overhead_under_one_percent() {
        use std::time::Instant;

        struct NoopStore;
        impl mesh_core_elements::VariableStore for NoopStore {
            fn get(&self, _: &str) -> Option<serde_json::Value> { None }
            fn keys(&self) -> Vec<String> { Vec::new() }
        }

        let iterations = 10_000usize;

        // Baseline: direct NoopStore get() calls
        let noop = NoopStore;
        let baseline_start = Instant::now();
        for _ in 0..iterations {
            let _ = mesh_core_elements::VariableStore::get(&noop, "audio.percent");
            let _ = mesh_core_elements::VariableStore::get(&noop, "volume");
            let _ = mesh_core_elements::VariableStore::get(&noop, "network.ssid");
        }
        let baseline_ns = baseline_start.elapsed().as_nanos().max(1);

        // Tracked: TrackingVariableStore wrapping NoopStore
        let tracking_start = Instant::now();
        for _ in 0..iterations {
            let t = TrackingVariableStore::new(&noop);
            let _ = mesh_core_elements::VariableStore::get(&t, "audio.percent");
            let _ = mesh_core_elements::VariableStore::get(&t, "volume");
            let _ = mesh_core_elements::VariableStore::get(&t, "network.ssid");
            let _ = t.into_reads(); // harvest
        }
        let tracking_ns = tracking_start.elapsed().as_nanos();

        let overhead_ratio = tracking_ns as f64 / baseline_ns as f64;
        assert!(
            overhead_ratio <= 1.01,
            "TrackingVariableStore overhead {:.4}x exceeds 1% threshold (baseline={}ns tracked={}ns)",
            overhead_ratio,
            baseline_ns,
            tracking_ns,
        );
    }
    ```

    Note: If this test is timing-sensitive on CI (slow machines), the 1.01
    threshold can be loosened to 1.05 with a comment explaining the intent is
    to detect catastrophic overhead (allocations per call, locks, etc.) not
    sub-microsecond variance. Prefer 1.01 on first pass; adjust only if CI fails.

    ```rust
    #[test]
    fn service_field_reads_populated_on_nodes() {
        // Build a representative tree: root with 3 children,
        // two of which have service field reads.
        let mut root = node(1, "column", 0.0, 0.0, 400.0, 200.0);

        let mut child_a = node(2, "text", 0.0, 0.0, 100.0, 20.0);
        child_a.service_field_reads.push(("audio".to_string(), "percent".to_string()));
        child_a.service_field_reads.push(("audio".to_string(), "muted".to_string()));

        let mut child_b = node(3, "text", 100.0, 0.0, 100.0, 20.0);
        // no service_field_reads on child_b

        let mut child_c = node(4, "icon", 200.0, 0.0, 20.0, 20.0);
        child_c.service_field_reads.push(("network".to_string(), "connected".to_string()));

        root.children.push(child_a);
        root.children.push(child_b);
        root.children.push(child_c);

        // Verify reads are present on the right nodes
        assert_eq!(root.children[0].service_field_reads.len(), 2);
        assert_eq!(root.children[0].service_field_reads[0], ("audio".to_string(), "percent".to_string()));
        assert_eq!(root.children[0].service_field_reads[1], ("audio".to_string(), "muted".to_string()));
        assert!(root.children[1].service_field_reads.is_empty());
        assert_eq!(root.children[2].service_field_reads.len(), 1);
        assert_eq!(root.children[2].service_field_reads[0], ("network".to_string(), "connected".to_string()));

        // Verify root itself has no reads (it wasn't given any)
        assert!(root.service_field_reads.is_empty());
    }
    ```
